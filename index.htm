    <script>

        const GAS_URL = 'https://script.google.com/macros/s/AKfycbyC9wI3SO8q15gPVZD6pKiwHDjDo1AswQKIX3K-6ktT7ISoJpqqhY4a8VnUq4e2NONm/exec';

       

        const OPEN_DATE = new Date('2025-12-21T23:00:00');

        const CLOSE_DATE = new Date('2025-12-28T11:30:00');

       

        const SESSIONS = [

            { id: 1, label: 'Session 1 (09:00-12:00, Jan 10, 2026)', screens: [120, 120, 90, 90] },

            { id: 2, label: 'Session 2 (14:00-18:00, Jan 10, 2026)', screens: [120, 120, 90, 90] },

            { id: 3, label: 'Session 3 (09:00-12:00, Jan 11, 2026)', screens: [120, 120, 90, 90] }

        ];

 

        const form = document.getElementById('avatar-form');

        const submitBtn = document.getElementById('submit-btn');

        const statusText = document.getElementById('form-status-text');

        const countdownEl = document.getElementById('countdown-timer');

        const sessionContainer = document.getElementById('session-container');

        const messageBox = document.getElementById('message-box');

 

        let serverData = null;

 

        function showMessage(msg, type = 'error') {

            messageBox.textContent = msg;

            messageBox.className = `message-box show ${type}`;

            setTimeout(() => messageBox.classList.remove('show'), 4000);

        }

 

        function updateTimers() {

            const now = new Date();

            let target = null;

            let status = "";

 

            if (now < OPEN_DATE) {

                target = OPEN_DATE;

                status = "FORM OPENS IN:";

                statusText.className = "text-lg font-bold text-amber-400 mb-1 uppercase";

                submitBtn.disabled = true;

            } else if (now < CLOSE_DATE) {

                target = CLOSE_DATE;

                status = "FORM IS OPEN:";

                statusText.className = "text-lg font-bold text-emerald-400 mb-1 uppercase";

                if(serverData) submitBtn.disabled = false;

            } else {

                status = "REGISTRATION CLOSED";

                statusText.className = "text-lg font-bold text-red-500 mb-1 uppercase";

                countdownEl.textContent = "See you next time!";

                submitBtn.disabled = true;

                statusText.textContent = status;

                return;

            }

 

            statusText.textContent = status;

            const diff = target - now;

            const d = Math.floor(diff / 86400000);

            const h = Math.floor((diff % 86400000) / 3600000);

            const m = Math.floor((diff % 3600000) / 60000);

            const s = Math.floor((diff % 60000) / 1000);

            countdownEl.textContent = `${d} Days : ${h} Hrs : ${m} Mins : ${s} Secs`;

        }

 

        async function init() {

            updateTimers();

            setInterval(updateTimers, 1000);

            await fetchData();

        }

 

        async function fetchData() {

            try {

                const controller = new AbortController();

                const timeoutId = setTimeout(() => controller.abort(), 8000);

               

                const res = await fetch(`${GAS_URL}?action=getStats`, { signal: controller.signal });

                const data = await res.json();

                clearTimeout(timeoutId);

               

                if (data.status === 'success') {

                    serverData = data.stats;

                    renderSessions(serverData);

                    const now = new Date();

                    if (now >= OPEN_DATE && now < CLOSE_DATE) submitBtn.disabled = false;

                }

            } catch (e) {

                console.error("Fetch error:", e);

                showMessage("Could not connect to server. Please refresh.");

                renderSessions({});

            }

        }

 

        function renderSessions(data) {

            sessionContainer.innerHTML = '';

            SESSIONS.forEach(session => {

                const group = document.createElement('div');

                group.innerHTML = `<h4 class="text-orange-400 font-black text-sm mb-2 border-l-4 border-orange-500 pl-2 uppercase">${session.label}</h4>`;

                const grid = document.createElement('div');

                grid.className = "grid grid-cols-1 gap-2";

 

                session.screens.forEach((capacity, idx) => {

                    const screenNum = idx + 1;

                    const key = `S${session.id}_R${screenNum}`;

                    const registered = data ? (data[key] || 0) : 0;

                    const remaining = capacity - registered;

                    const isSoldOut = remaining <= 0;

 

                    const label = document.createElement('label');

                    label.className = `radio-card flex justify-between items-center p-3 rounded-xl cursor-pointer ${isSoldOut ? 'sold-out' : ''}`;

                    label.innerHTML = `

                        <div class="flex items-center">

                            <input type="radio" name="sessionScreen" value="${key}"

                                   data-session-label="${session.label}"

                                   data-screen="Screen ${screenNum}"

                                   required

                                   class="mr-3 accent-orange-500" ${isSoldOut ? 'disabled' : ''}>

                            <div class="text-xs uppercase font-bold">Screen ${screenNum}</div>

                        </div>

                        <div class="text-right">

                            <span class="${isSoldOut ? 'text-red-500 font-black' : 'text-slate-300'} font-mono text-sm">

                                ${isSoldOut ? 'SOLD OUT' : remaining + ' left'}

                            </span>

                        </div>

                    `;

                    grid.appendChild(label);

                });

                group.appendChild(grid);

                sessionContainer.appendChild(group);

            });

        }

 

        form.addEventListener('submit', async (e) => {

            e.preventDefault();

           

            const nameInput = document.getElementById('full-name');

            nameInput.value = nameInput.value.trim().split(' ').map(w => w.charAt(0).toUpperCase() + w.slice(1).toLowerCase()).join(' ');

 

            const selectedRadio = document.querySelector('input[name="sessionScreen"]:checked');

            if (!selectedRadio) {

                showMessage("Please select a session and screen!");

                return;

            }

           

            const sessionLabel = selectedRadio.dataset.sessionLabel;

            const screenLabel = selectedRadio.dataset.screen;

 

            const formData = new FormData(form);

            const payload = Object.fromEntries(formData);

           

            submitBtn.disabled = true;

            document.getElementById('spinner').classList.remove('hidden');

 

            try {

                const res = await fetch(GAS_URL, {

                    method: 'POST',

                    body: JSON.stringify({

                        ...payload,

                        action: 'register',

                        sessionFull: sessionLabel,

                        screenFull: screenLabel

                    })

                });

                const result = await res.json();

               

                if (result.status === 'success') {

                    document.getElementById('registration-content').classList.add('hidden');

                    document.getElementById('success-screen').classList.remove('hidden');

                    document.getElementById('res-name').textContent = payload.fullName;

                    document.getElementById('res-class').textContent = payload.class.toUpperCase();

                    document.getElementById('res-session').textContent = sessionLabel;

                    document.getElementById('res-screen').textContent = screenLabel;

                    window.scrollTo(0, 0);

                } else {

                    showMessage(result.message || "Registration failed. Please try again.");

                    submitBtn.disabled = false;

                }

            } catch (err) {

                showMessage("Connection error or session full. Please refresh.");

                submitBtn.disabled = false;

            } finally {

                document.getElementById('spinner').classList.add('hidden');

            }

        });

 

        init();

    </script>
